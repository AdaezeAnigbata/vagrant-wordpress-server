Vagrant.configure("2") do |config|
  config.vm.box = "ubuntu/focal64"
  config.vm.network "private_network", ip: "192.168.56.30"

  config.vm.provision "shell", inline: <<-SHELL
    # 1. Kill and remove Apache completely
    systemctl stop apache2
    apt-get remove -y apache2 apache2-utils apache2-bin apache2.2-common
    apt-get autoremove -y

    # 2. Update and install Nginx
    apt-get update
    apt-get install -y nginx

    # 3. Create the simple HTML page for the assignment
    echo "<!DOCTYPE html><html><head><title>Adeze Nginx Assignment</title></head><body><h1>Success!</h1><p>This is my simple HTML page served by Nginx.</p></body></html>" > /var/www/html/index.html

    # 4. Start Nginx to take over Port 80
    systemctl restart nginx
  SHELL
end
